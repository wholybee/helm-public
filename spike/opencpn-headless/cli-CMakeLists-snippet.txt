# ---- Helm Phase-1 spike: headless nav-core probe (mirrors opencpn-cmd) ----
add_executable(helm-spike helm_spike.cpp api_shim.cpp)
set_property(TARGET helm-spike PROPERTY ENABLE_EXPORTS ON)
target_include_directories(helm-spike PRIVATE
  ${PROJECT_SOURCE_DIR}/include ${PROJECT_SOURCE_DIR}/gui/include)
target_compile_definitions(helm-spike PUBLIC USE_MOCK_DEFS)
if (TARGET ocpn::wxcurl)
  target_link_libraries(helm-spike PRIVATE ocpn::wxcurl)
else ()
  target_link_libraries(helm-spike PRIVATE ${SYS_WXCURL_LIBRARIES})
endif ()
target_link_libraries(helm-spike PRIVATE ${wxWidgets_LIBRARIES})
if (TARGET ocpn::wxsvg)
  target_link_libraries(helm-spike PRIVATE ocpn::wxsvg)
endif ()
target_link_libraries(helm-spike PRIVATE ocpn::serial ocpn::tinyxml ocpn::filesystem ocpn::gl-headers)
if (HAVE_LIBUDEV)
  target_link_libraries(helm-spike PRIVATE ocpn::libudev)
endif ()
target_link_libraries(helm-spike PRIVATE ocpn::model ocpn::model-src)
